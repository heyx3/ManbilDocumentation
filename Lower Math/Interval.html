Usage

This class can be used to represent spans across a scalar space. It is immutable; there are no public fields or non-const functions.

Interval oneToThree(1.0f, 3.0f, 0.0f, true, true);
std::cout << "The start of the interval is " << oneToThree.GetStart() << // 1.0
            "\nThe end of the interval is " << oneToThree.GetEnd() << // 3.0
            "\nThe range of the interval is " << oneToThree.GetRange(); // 2.0

float third = oneToThree.Lerp(0.333333f);
std::cout << "A third of the way through the interval is the value " << third << "\n"; // 1.666666

Interval zeroToFour = oneToThree.Widen(2.0f), // Add 2.0 to the range of the interval.
        tenToFourteen = zeroToFour.Translate(10.0f), // Move the interval over by 10.0.
        eightToSixteen = tenToFourteen.Inflate(2.0f), // Double the interval's range.
        zeroToSixteen = Interval::Union(zeroToFour, eightToSixteen), // Gets the union of the two.
        negOneToSixteen = Interval::Union(zeroToSixteen, zeroToFour.Translate(-1.0f)), // Another union.
        invalidInterval = Interval::Intersection(zeroToFour, tenToFourteen); // An Interval that doesn't span any space.
